let h_gra, a_gra;

do {
    let liczba1 = prompt("Wysokość = ", "");
    h_gra = parseFloat(liczba1);
} while (h_gra <= 0);

do {
    let liczba2 = prompt("Podstawa = ", "");
    a_gra = parseFloat(liczba2);
} while (!(a_gra > 0));

let Pole = (a_gra * h_gra) / 2;
document.write("Pole = " + Pole.toFixed(2) + " cm²");
